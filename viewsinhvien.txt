        DefaultTableModel tblModdel;
        
        public void initTable(){
            tblModdel = (DefaultTableModel) bangTable.getModel();
            String[] colun = new String[]{
              "Mã sinh viên","Tên sinh viên","Ngày sinh","Tổng điểm"  
            };
            tblModdel.setColumnIdentifiers(colun);
        }
        public void fillTable(){
            tblModdel = (DefaultTableModel) bangTable.getModel();
            tblModdel.setRowCount(0);
            
            for (SinhVien sv : list) {
                Object[] row = new Object[4];
                row[0] = sv.getMaSinhVien();
                row[1] = sv.getHoTen();
                row[2] = sv.getNgaySinh();
                row[3] = sv.getTONGDIEM();
              
                tblModdel.addRow(row);
            }
        }
        
        public SinhVien getModel(){
            SinhVien sv = new SinhVien();
            sv.setMaSinhVien(maTxt.getText());
           sv.setHoTen(hotenTxt.getText());
           sv.setNgaySinh(ngaysinhTxt.getText());
           sv.setTONGDIEM(tongdiemTxt.getText());
           return sv;
        }
        
        public void setModel(SinhVien s){
            maTxt.setText(s.getMaSinhVien());
            hotenTxt.setText(s.getHoTen());
            ngaysinhTxt.setText(s.getNgaySinh());
            tongdiemTxt.setText(s.getTONGDIEM());
        }

    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {                                         
       // Thêm sinh viên mới vào cơ sở dữ liệu

    list = sv.getAllDao();
    fillTable();
    JOptionPane.showMessageDialog(this, "Thêm Thành Công");
    }                                        

    private void bangTableMouseClicked(java.awt.event.MouseEvent evt) {                                       
        // TODO add your handling code here:
        int chon = bangTable.getSelectedRow();
        setModel(list.get(chon));
    }                                      

    private void jButton2ActionPerformed(java.awt.event.ActionEvent evt) {                                         
        // TODO add your handling code here:
        sv.update(getModel());
        list = sv.getAllDao();
        fillTable();
        JOptionPane.showMessageDialog(this, "Sửa Thành Công");
    }                                        

    private void jButton3ActionPerformed(java.awt.event.ActionEvent evt) {                                         
        // TODO add your handling code here:
         // Lấy dòng đã chọn từ bảng
    int chon = bangTable.getSelectedRow();

    if (chon != -1) { // Nếu có dòng được chọn
        SinhVien s = list.get(chon); // Lấy đối tượng sinh viên từ danh sách
        sv.delete(s); // Xóa sinh viên trong cơ sở dữ liệu
        list = sv.getAllDao(); // Cập nhật lại danh sách sinh viên
        fillTable(); // Cập nhật lại bảng
        JOptionPane.showMessageDialog(this, "Xóa Thành Công"); // Thông báo
    } else {
        JOptionPane.showMessageDialog(this, "Vui lòng chọn sinh viên để xóa!"); // Nếu không chọn sinh viên nào
    }
    }
